  <body>
    <div id='map'></div>
    <script type="text/javascript">

      var map = L.map('map', {
        zoomControl: false,
        center: [48.81485697596884, 2.230295999770],
        zoom: 12,
        subdomains: [],
        minZoom: 12,
        maxZoom: 20
      });

      // map.scrollWheelZoom.disable();

      L.tileLayer('http://{s}.tile.osm.org/{z}/{x}/{y}.png', {
        minZoom: 12,
        maxZoom: 20,
        attribution: '&copy; <a href="http://osm.org/copyright">OpenStreetMap</a> contributors'
      }).addTo(map);

      var igkLayer = L.tileLayer('http://127.0.0.1:9000/tiles/{z}/{x}/{y}.png', {
        attribution: '(C) 2017 IGK'
      }).addTo(map);

      var opacitySlider = new L.Control.opacitySlider();
      map.addControl(opacitySlider);
      opacitySlider.setOpacityLayer(igkLayer);

    </script>
